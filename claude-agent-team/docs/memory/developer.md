## TypeScript 배열 중복 제거 제네릭 함수

### 구현

```typescript
// 기본 타입(primitive) 전용 - Set 기반
function unique<T>(arr: T[]): T[] {
  return [...new Set(arr)];
}

// 객체 포함 - 키 추출 함수 기반
function uniqueBy<T, K>(arr: T[], keyFn: (item: T) => K): T[] {
  const seen = new Map<K, T>();
  for (const item of arr) {
    const key = keyFn(item);
    if (!seen.has(key)) {
      seen.set(key, item);
    }
  }
  return [...seen.values()];
}

// 커스텀 동등성 비교 - 완전한 유연성 제공
function uniqueWith<T>(arr: T[], isEqual: (a: T, b: T) => boolean): T[] {
  return arr.filter((item, index) =>
    arr.findIndex((other) => isEqual(item, other)) === index
  );
}
```
